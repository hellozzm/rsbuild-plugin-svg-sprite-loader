## Usage

Install:

```bash
npm add rsbuild-svg-sprite-loader -D
```

Add plugin to your `rsbuild.config.ts`:

```ts
// rsbuild.config.ts
import { pluginSvgSpriteLoader } from "rsbuild-plugin-svg-sprite-loader";

export default {
  plugins: [
    pluginSvgSpriteLoader({
      path: join(__dirname, 'src/icons'),
      symbolId: 'icon-[name]',
    }),
  ],
};
```
